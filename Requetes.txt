1- Les noms des neveux de Ahmed

PREFIX fam: <http://www.semanticweb.org/ontologies/famille#>

SELECT DISTINCT ?nom WHERE {
    ?enfantAhmed fam:enfantDe fam:Ahmed .
    ?frere fam:enfantDe fam:Ahmed .
    FILTER(?frere != ?enfantAhmed)
    ?neveu fam:enfantDe ?frere .
    ?neveu fam:nom ?nom .
}


2-Les noms des frères

PREFIX fam: <http://www.semanticweb.org/ontologies/famille#>

SELECT DISTINCT ?nom1 ?nom2 WHERE {
    ?x fam:enfantDe ?parent .
    ?y fam:enfantDe ?parent .
    ?x a fam:Homme .
    ?y a fam:Homme .
    FILTER(?x != ?y)
    ?x fam:nom ?nom1 .
    ?y fam:nom ?nom2 .
}

6- Les noms des cousins et cousines de Samia

PREFIX fam: <http://www.semanticweb.org/ontologies/famille#>

SELECT DISTINCT ?nom WHERE {
    fam:Samia fam:enfantDe ?parentSamia .
    ?oncleOuTante fam:enfantDe ?grandParent .
    ?parentSamia fam:enfantDe ?grandParent .
    FILTER(?oncleOuTante != ?parentSamia)
    ?cousin fam:enfantDe ?oncleOuTante .
    FILTER(?cousin != fam:Samia)
    ?cousin fam:nom ?nom .
}
